     public boolean updatePart(int id, Part part){
         int index = -1;
         for(Part partId : Inventory.getAllParts()){
             index++;
             if(partId.getId() == id){
                 Inventory.getAllParts().set(index, part);
                 return true;
             }
         }
         return false;
     }
 
     public boolean deletePart(int id){
         for(Part partId : Inventory.getAllParts()){
             if(partId.getId() == id)
                 return Inventory.getAllParts().remove(partId);
         }
         return false;
     }
